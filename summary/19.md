# 19. Remove Nth Node From End of List

## [Description](https://leetcode.com/problems/remove-nth-node-from-end-of-list/)

## Solution

```go
// Объявление структуры односвязного списка
type ListNode struct {
     Val int
     Next *ListNode
}
```

```go
// Ключевая идея:
// Поменять удаление с конца, на удаление с начала. Узнать длину списка.
// Рассчитать координату с начала списка для узла перед удаляемым.
// Time Complexity: O(2n)
// Space Complexity: O(1)
func removeNthFromEnd(head *ListNode, n int) *ListNode {
    // строим новый список и не знаем заранее его голову, так как она может быть
    // удалена. создаем Dummy Node.
    dummy := &ListNode{Next: head}
    // считаем длину списка с учетом Dummy Node.
    cur := dummy
    length := 0
    for cur != nil {
        cur = cur.Next
        length++
    }
    // считаем координату с начала списка для узла перед удаляемым.
    pos := length - n - 1
    cur = dummy
    // доходим до узла перед удаляемым
    for range pos {
        cur = cur.Next
    }
    // удаляем узел
    cur.Next = cur.Next.Next
    return dummy.Next
}
```

```go
// Ключевая идея:
// Взять два указателя. Отодвинуть первый на ожидаемый размер n. Затем двигать
// оба указателя, пока первый не укажет на nil.
// Time Complexity: O(n)
// Space Complexity: O(1)
func removeNthFromEnd(head *ListNode, n int) *ListNode {
    // строим новый список и не знаем заранее его голову, так как она может быть
    // удалена. создаем Dummy Node.
    dummy := &ListNode{Next: head}
    // объявляем два указателя
    first, second := dummy, dummy
    // двигаем первый на ожидаемый размер n с учетом Dummy Node
    for range n+1 {
        first = first.Next
    }
    // двигаем оба указателя, пока первый не укажет на nil
    for first != nil {
        first = first.Next
        second = second.Next
    }
    // удаляем узел
    second.Next = second.Next.Next
    return dummy.Next
}
```

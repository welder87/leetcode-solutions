# 2095. Delete the Middle Node of a Linked List

## [Description](https://leetcode.com/problems/delete-the-middle-node-of-a-linked-list/)

## Solution

```go
type ListNode struct {
     Val int
     Next *ListNode
}

// Ключевая идея:
// Использовать быстрый и медленный указатели, пока быстрый указатель и
// следующий за ним не nil.
// При этом сохраняем указатель на предыдущий медленному.
// Это даст возможность удалить далее середину.
func deleteMiddle(head *ListNode) *ListNode {
    // граничное условие для отсутствия списка или одного узла в списке
    if head == nil || head.Next == nil {
        return nil
    }
    // инициализация трех указателей головой списка
    slow, fast, prev := head, head, head
    // условие цикла: пока быстрый указатель и следующий за ним не nil
    for fast != nil && fast.Next != nil {
         // сдвигаем на 2 узла вперед
        fast = fast.Next.Next
        // сохраняем текущий медленный указатель
        prev = slow
        // сдвигаем на 1 узел вперед
        slow = slow.Next
    }
    // разрываем связь между серединой и остальными узлами
    prev.Next = prev.Next.Next
    return head
}
```

```go
type ListNode struct {
     Val int
     Next *ListNode
}

// Ключевая идея:
// Использовать быстрый и медленный указатели, пока быстрый указатель и
// следующий за ним не nil.
// Но для медленного указателя начать с Dummy Node.
// Это даст возможность найти узел предыдущий перед серединой.
func deleteMiddle(head *ListNode) *ListNode {
    dummy := &ListNode{Next: head}
    slow, fast := dummy, head
    for fast != nil && fast.Next != nil {
        fast = fast.Next.Next
        slow = slow.Next
    }
    slow.Next = slow.Next.Next
    return dummy.Next
}
```

# 141. Linked List Cycle

## [Description](https://leetcode.com/problems/linked-list-cycle/)

## Solution

```go
type ListNode struct {
     Val int
     Next *ListNode
}
// Ключевая идея:
// Использовать быстрый и медленный указатели, пока быстрый указатель и
// следующий за ним не nil. Если в какой-то момент быстрый и медленный указатели
// равны - значит цикл есть.
func hasCycle(head *ListNode) bool {
    // инициализация указателей головой списка
    slow, fast := head, head
    // условие цикла: пока быстрый указатель и следующий за ним не nil
    for fast != nil && fast.Next != nil {
        // сдвигаем на 2 узла вперед
        fast = fast.Next.Next
        // сдвигаем на 1 узел вперед
        slow = slow.Next
        // проверяем равенство указателей
        if fast == slow {
            return true
        }
    }
    return false
}
```

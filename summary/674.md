# 674. Longest Continuous Increasing Subsequence

## [Description](https://leetcode.com/problems/find-pivot-index/)

## Solution

```go
import "math"

func findLengthOfLCIS(nums []int) int {
    // инициализируем переменные
    // максимальный размер подмассива,
    // текущий размера подмассива,
    // предыдущий элемент Sentinel value - считаем, что система 64 битная.
    ans, cur, prev := 0, 0, math.MinInt64
    // итеративно проходим по массиву от начала до конца
    for _, num := range nums {
        // сравниваем текущий элемент с предыдущим
        if num > prev {
            // в случае соблюдения условия - инкрементируем текущий размера
            // подмассива
            cur++
        } else {
            // сбрасываем счетчик размера подмассива
            cur = 1
        }
        // присваиваем текущий элемент в предыдущий для следующей итерации
        prev = num
        // упрощаем себе жизнь, избавляясь от граничной обработки после
        // остановки цикла
        ans = max(cur, ans)
    }
    return ans
}
```

```go
// Используем два указателя для контроля позиций одинаковых символов.
// С помощью подцикла находим подмассива через разницу указателей. Cдвигаем
// левый указатель на правый, а правый инкрементируем после выхода из подцикла.
func findLengthOfLCIS(nums []int) int {
    i, j := 0, 1
    ans := 1
    for j < len(nums) {
        // инкрементируем левый указатель до тех пор пока текущий элемент больше
        // следующего либо массив закончен
        for j < len(nums) && nums[j] > nums[j - 1] {
            j++
        }
        // вычисляем разницу через указатели и сравниваем с максимальным
        // значением подмассива
        ans = max(j - i, ans)
        // сдвигаем левый указатель на правый, а правый инкрементируем
        i = j
        j++
    }
    return ans
```

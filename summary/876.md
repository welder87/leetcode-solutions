# 876. Middle of the Linked List

## [Description](https://leetcode.com/problems/middle-of-the-linked-list/)

## Solution

```go
type ListNode struct {
     Val int
     Next *ListNode
}
// Ключевая идея:
// Использовать быстрый и медленный указатели, пока быстрый указатель и
// следующий за ним не nil
func middleNode(head *ListNode) *ListNode {
    // инициализация указателей головой списка
    fast, slow := head, head
    // условие цикла: пока быстрый указатель и следующий за ним не nil
    for fast != nil && fast.Next != nil {
        // сдвигаем на 2 узла вперед
        fast = fast.Next.Next
        // сдвигаем на 1 узел вперед
        slow = slow.Next
    }
    return slow
}
```

```go
// То же самое решение, но с DummyNode.
// Помогает для поиска удаляемого узла.
func middleNode(head *ListNode) *ListNode {
    dummy := &ListNode{Next: head}
    fast, slow := head, dummy
    for fast != nil && fast.Next != nil {
        fast = fast.Next.Next
        slow = slow.Next
    }
    return slow.Next
}
```

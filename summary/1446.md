# 1446. Consecutive Characters

## [Description](https://leetcode.com/problems/consecutive-characters/)

## Solution

```go
// Ключевая идея:
// Проходим по строке и сравниваем текущий с предыдущим, параллельно
// инкрементируем счетчик количества одинаковых символов. Сбрасываем,если
// текущий и предыдущий неодинаковы.
func maxPower(s string) int {
    // инициализируем переменные
    // максимальный размер подстроки,
    // текущий размера подстроки,
    ans, cur := 0, 0
    // предыдущий элемент Sentinel value - zero value для руны
    var prev rune
    // итеративно проходим по строке (массиву рун) от начала до конца
    for _, sym := range s {
        // сравниваем текущий элемент с предыдущим
        if prev == sym {
            // в случае соблюдения условия - инкрементируем текущий размера
            // подстроки
            cur++
        } else {
            // сбрасываем счетчик размера подстроки
            cur = 1
        }
        // присваиваем текущий элемент в предыдущий для следующей итерации
        prev = sym
        // упрощаем себе жизнь, избавляясь от граничной обработки после
        // остановки цикла
        ans = max(ans, cur)
    }
    return ans
}
```

```go
// Используем два указателя для контроля позиций одинаковых символов.
// С помощью подцикла находим количество одинаковых символов через разницу
// указателей. Cдвигаем левый указатель на правый, а правый инкрементируем после
// выхода из подцикла.
func maxPower(s string) int {
    // В этой реализации для go есть существенный минус. В случае если в строке
    // есть не только ASCII, стоит выделить дополнительную память и
    // преобразовать в массив рун.
    ar := []rune(s)
    i, j := 0, 1
    mx := 1
    for j < len(ar) {
        // инкрементируем левый указатель до тех пор пока значения по указателям
        // равны либо массив закончен
        for j < len(ar) && ar[j] == ar[i] {
            j++
        }
        // вычисляем разницу через указатели и сравниваем с максимальным
        // значением подстроки
        mx = max(j - i, mx)
        // сдвигаем левый указатель на правый, а правый инкрементируем
        i = j
        j++
    }
    return mx
}
```

# 3819. Rotate Non Negative Elements

## [Description](https://leetcode.com/problems/rotate-non-negative-elements/)

## Solution

```go
// Ключевая идея:
// Достаем не негативные элементы. Делаем их левый поворот на k,по стандартному
// паттерну: реверс от 0 до k-1, реверс от k до len-1, полный реверс. Заполняем
// в порядке движения по основному массиву не негативные элементы.
// Time Complexity: O(n) или o(n + 2m + n)
// Space Complexity: O(n) или o(m)
func rotateElements(nums []int, k int) []int {
    nonNeg := make([]int, 0, len(nums))
    for i := range len(nums) {
        if nums[i] >= 0 {
            nonNeg = append(nonNeg, nums[i])
        }
    }
    if len(nonNeg) <= 1 {
        return nums
    }
    rotate(nonNeg, k)
    j := 0
    for i := range len(nums) {
        if nums[i] >= 0 {
            nums[i] = nonNeg[j]
            j++
        }
    }
    return nums
}

func rotate(ar []int, k int)  {
    if k > len(ar) {
        k %= len(ar)
    }
    reverse(ar, 0, k - 1)
    reverse(ar, k, len(ar) - 1)
    reverse(ar, 0, len(ar) - 1)
}

func reverse(ar []int, l, r int) {
    for l < r {
        ar[l], ar[r] = ar[r], ar[l]
        l++
        r--
    }
}
```

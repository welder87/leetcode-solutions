# 206. Reverse Linked List

## [Description](https://leetcode.com/problems/reverse-linked-list/description/)

## Solution

```go
// Решение с доп переменной в цикле
func reverseList(head *ListNode) *ListNode {
    // указатель на предыдущий - ставим в nil
    var prev *ListNode
    // указатель на текущий - ставим в head
    cur := head
    // условие цикла: пока указатель на текущий элемент не nil.
    for cur != nil {
        // копируем указатель на текущий элемент во временную переменную
        tmp := cur
        // двигаем указатель на следующий элемент, чтобы сохранить ссылку
        // при изменении далее
        cur = cur.Next
        // разрываем связь между текущим элементом и следующим
        tmp.Next = prev
        // текущим элемент становится предыдущим
        prev = tmp
    }
    // возвращаем указатель на предыдущий элемент
    return prev
}
```

# 1721. Swapping Nodes in a Linked List

## [Description](https://leetcode.com/problems/swapping-nodes-in-a-linked-list/)

## Solution

```go
type ListNode struct {
     Val int
     Next *ListNode
}
// Ключевая идея:
// Найти k-й элемент с начала через прямую итерацию, одновременно узнав длину
// списка. Рассчитать (длина - k)-й с для узла с конца.
// Time Complexity: O(2n)
// Space Complexity: O(1)
func swapNodes(head *ListNode, k int) *ListNode {
    // строим новый список и не знаем заранее его голову, так как она может быть
    // удалена. создаем Dummy Node.
    dummy := &ListNode{Next: head}
    // указатель на k-й узел с начала
    var left *ListNode
    // считаем длину списка с учетом Dummy Node.
    cur := dummy
    length := 0
    // условие цикла: пока текущий указатель не nil
    for cur != nil {
        cur = cur.Next
        length++
        if length == k {
            // находим указатель на k-й узел с начала
            left = cur
        }
    }
    // указатель на k-й узел с конца
    right := dummy
    // условие цикла: пока не закончится итератор от (длина - k)
    for range length - k {
        right = right.Next
    }
    // меняем значения в найденных узлах
    left.Val, right.Val = right.Val, left.Val
    return head
}
```

```go
type ListNode struct {
     Val int
     Next *ListNode
}
// Ключевая идея:
// Найти k-й элемент с начала через прямую итерацию, одновременно узнав длину
// списка. Рассчитать (длина - k)-й с для узла с конца.
// Time Complexity: O(2n)
// Space Complexity: O(1)
func swapNodes(head *ListNode, k int) *ListNode {
    // строим новый список и не знаем заранее его голову, так как она может быть
    // удалена. создаем Dummy Node.
    dummy := &ListNode{Next: head}
    cur := dummy
    // двигаем указатель на ожидаемый размер k с учетом Dummy Node
    for range k {
        cur = cur.Next
    }
    // объявляем два указателя
    first, second := cur, dummy
    // двигаем оба указателя, пока первый не укажет на nil
    for cur != nil {
        cur = cur.Next
        second = second.Next
    }
    // меняем значения в найденных узлах
    first.Val, second.Val = second.Val, first.Val
    return head
}
```

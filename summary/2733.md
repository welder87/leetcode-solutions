# 2733. Neither Minimum nor Maximum

## [Description](https://leetcode.com/problems/neither-minimum-nor-maximum/)

## Solution

```go
// Ключевая идея:
// Находим первый и второй максимумы.
// Второй элемент, если он есть, всегда будет валидным значением.
// Time Complexity: O(n)
// Space Complexity: O(1)
func findNonMinOrMax(nums []int) int {
    if len(nums) <= 2 {
        return -1
    }
    first, second := -1, -1
    for _, num := range nums {
        if first < num {
            second, first = first, num
        } else {
            second = max(second, num)
        }
    }
    return second
}
```

```go
// Ключевая идея:
// Используем Ordered Set для хранения уникальных элементов в порядке убывания.
// Второй элемент, если он есть, всегда будет валидным значением.
// Time Complexity: O(n)
// Space Complexity: O(1) или o(2)
func findNonMinOrMax(nums []int) int {
    if len(nums) <= 2 {
        return -1
    }
    osMax := GetOrderedSet()
    for _, num := range nums {
        osMax.Push(num)
    }
    if second := osMax.Second(); second == 0 {
        return -1
    } else {
        return second
    }
}

type OrderedSet struct{
    nums [2]int
}

func GetOrderedSet() *OrderedSet {
    // используем Sentinel Value для инициализации массива
    return &OrderedSet{}
}

func (s *OrderedSet) Push(val int) {
    if s.nums[0] < val {
        s.nums[1], s.nums[0] = s.nums[0], val
    } else if s.nums[1] < val {
        s.nums[1] = val
    }
}

func (s *OrderedSet) Second() int {
    return s.nums[1]
}
```

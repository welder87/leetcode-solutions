# 2405. Optimal Partition of String

## [Description](https://leetcode.com/problems/optimal-partition-of-string/)

## Solution

```go
// Time Complexity: O(n)
// Space Complexity: O(m)
func partitionString(s string) int {
    ans := 0
    // преобразуем в массив рун для верной адресации по индексу
    ar := []rune(s)
    // инициализируем левый и правый указатели на один элемент
    i, j := 0, 1
    // Hash Set с уникальными символами
    un := make(map[rune]struct{})
    // проверяем в цикле граничное условие окончания массива для левого
    // указателя
    for i < len(ar) {
        //кладем i-тый элемент
        un[ar[i]] = struct{}{}
        // проверяем в цикле граничное условие окончания массива для правого
        // указателя
        for j < len(ar) {
            // проверяем в цикле наличие элемента в Hash Set
            if _, ok := un[ar[j]]; !ok {
                // если нет - заполняем Hash Set
                un[ar[j]] = struct{}{}
            } else {
                // если есть останавливаем цикл
                break
            }
            j++
        }
        // инкрементируем счетчик
        ans++
        // очищаем Hash Set
        un = make(map[rune]struct{})
        // сдвигаем левый указатель на место правого
        i = j
        // правый указатель
        j++
    }
    return ans
}
```

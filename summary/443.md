# 443. String Compression

## [Description](https://leetcode.com/problems/string-compression/)

## Solution

```go
import "strconv"

// Ключевая идея:
// Фиксируем левый указатель и двигаем правый пока символы под ними совпадают.
// Time Complexity: O(n)
// Space Complexity: O(1)
func compress(chars []byte) int {
    // инициализируем левый и правый указатели на один элемент
    // добавляем счетчик, который одновременно и будет ответом
    i, j, ans := 0, 0, 0
    for j < len(chars) {
        // кладем по месту символ как первый элемент в паттерне %символ-число%
        chars[ans] = chars[i]
        // инкрементируем счетчик
        ans++
        // двигаем правый указатель пока символы под левым и правым совпадают
        for j < len(chars) && chars[i] == chars[j] {
            j++
        }
        // считаем разницу указателей
        cnt := j - i
        // если разница указателей больше единицы, то нужно добавлять символы
        // количества
        if cnt > 1 {
            // переводим в строку
            num := strconv.Itoa(cnt)
            // движемся по строке заполняя массив символов
            for k := range num {
                chars[ans] = num[k]
                // инкрементируем счетчик
                ans++
            }
        }
        // сдвигаем левый указатель
        i = j
    }
    return ans
}
```

# 228. Summary Ranges

## [Description](https://leetcode.com/problems/summary-ranges/description/)

## Solution

```go
import "strconv"

// Ключевая идея:
// Массив ужн отсортирован и с уникальными элементами. Проверяем два соседних
// элемента на разницу в единицу. Двигаем так дальше, пока условие соблюдается.
// Time Complexity: O(n)
// Space Complexity: O(n)
func summaryRanges(nums []int) []string {
    ans := []string{}
    // инициализируем левый и правый указатели на элементы рядом друг с другом
    i, j := 0, 1
    // проверяем в цикле граничное условие окончания массива для левого
    // указателя
    for i < len(nums) {
        // проверяем в цикле граничное условие окончания массива для правого
        // указателя, разницу между соседними элементами.
        for j < len(nums) && nums[j] - nums[j - 1] == 1 {
            // двигаем правый указатель
            j++
        }
        // собираем результат
        ans = append(ans, makeRange(nums, i, j - 1))
        // сдвигаем границы
        i = j
        j++
    }
    return ans
}

func makeRange(nums []int, l, r int) string {
    if r == l {
        return strconv.Itoa(nums[l])
    }
    return strconv.Itoa(nums[l]) + "->" + strconv.Itoa(nums[r])
}
```

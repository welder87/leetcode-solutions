# 2574. Left and Right Sum Differences

## [Description](https://leetcode.com/problems/left-and-right-sum-differences)

## Solution

```go
// Ключевая идея:
// Считаем префиксную и постфиксную суммы по ходу движения по массиву и вычитаем
// разницу.
func leftRightDifference(nums []int) []int {
    // сначала рассчитаем сумму всех чисел в массиве и положим в переменную для
    // постфиксной суммы
    r := 0
    for _, num := range nums {
        r += num
    }
    // префиксную сумму инициализируем как ноль.
    l := 0
    // создаем массив по размеру равный исходному
    ans := make([]int, len(nums))
    // двигаемся по индексам массива
    for i, num := range nums {
        // сначала вычитаем постфиксной суммы текущий элемент
        r -= num
        // рассчитываем разность по модулю и кладем по порядковому индексу
        ans[i] = abs(r - l)
        // добавляем к префиксной сумме текущий элемент
        l += num
    }
    return ans
}

func abs(num int) int {
    if num < 0 {
        return num * -1
    }
    return num
}
```

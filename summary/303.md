# 303. Range Sum Query - Immutable

## [Description](https://leetcode.com/problems/range-sum-query-immutable/)

## Solution

```go
type NumArray struct {
    nums []int
}

// Time Complexity: O(n)
// Space Complexity: O(n)
func Constructor(nums []int) NumArray {
    // используем стандартный паттерн
    // выделяем префиксный массив размером N+1, где элемент с индексом 0 всегда
    // равен нулю
    px := make([]int, len(nums) + 1)
    // проходим, заполняя суммы с первого индекса
    for i, num := range nums {
        px[i+1] = px[i] + num
    }
    return NumArray{nums: px}
}

// Time Complexity: O(1)
// Space Complexity: O(1)
func (this *NumArray) SumRange(left int, right int) int {
    // расчет суммы для отрезка [left, right]
    return this.nums[right+1] - this.nums[left]
}
```

```go
type NumArray struct {
    nums []int
}

// Решение без использования нового массива
// Time Complexity: O(n)
// Space Complexity: O(1)
func Constructor(nums []int) NumArray {
    for i, num := range nums[1:] {
        nums[i+1] = nums[i] + num
    }
    return NumArray{nums: nums}
}

// Time Complexity: O(1)
// Space Complexity: O(1)
func (this *NumArray) SumRange(left int, right int) int {
    val := 0
    if left > 0 {
        val = this.nums[left - 1]
    }
    return this.nums[right] - val
}
```

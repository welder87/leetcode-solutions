# 328. Odd Even Linked List

```go
type ListNode struct {
     Val int
     Next *ListNode
}
 // Ключевая идея:
 // Создаем две списка - указатели на четные и нечетные узлы. Разбираем узлы
 // списка на два вышеуказанных и потом соединяем последовательно.
 // Time Complexity: O(n)
 // Space Complexity: O(1)
func oddEvenList(head *ListNode) *ListNode {
    // граничное условие, если узлов нет.
    if head == nil {
        return nil
    }
    // два указателя c Dummy Node для четных и нечетных узлов
    odd, even := &ListNode{}, &ListNode{}
    // сохраняем головы указателей для четных и нечетных узлов
    oddHead, evenHead := odd, even
    // счетчик для определения чет/нечет
    counter := 1
    // условие цикла: пока указатель на узел не nil.
    for head != nil {
        // разрываем связь у текущего узла
        tmp := head.Next
        head.Next = nil
        // в зависимости от четности вставляем в нужный список
        if counter%2 == 0 {
            even.Next = head
            even = even.Next
        } else {
            odd.Next = head
            odd = odd.Next
        }
        // двигаемся дальше
        head = tmp
        counter++
    }
    // соединяем списки
    oddHead = oddHead.Next
    odd.Next = evenHead.Next
    return oddHead
}
```

# 234. Palindrome Linked List

## [Description](https://leetcode.com/problems/palindrome-linked-list/)

## Solution

```go
type ListNode struct {
     Val int
     Next *ListNode
}

// Ключевая идея:
// Находим середину списка, от неё делаем реверс, а затем в цикле идем по
// каждому узлу двух получившихся списков.
func isPalindrome(head *ListNode) bool {
    // Находим середину списка.
    // Используем код задачи 876.
    mid := middleNode(head)
    // Делаем реверс от середины списка.
    // Используем код задачи 206.
    rev := reverseList(mid)
    // Условие цикла: пока указатель на текущий узел списка с реверсом
    // не nil.
    // Этого достаточно, так как в случае четного количества узлов, у двух
    // получаемых списков количество узлов равно, а в случае нечетного,
    // оставшийся узел будет всегда привязан к первому, то есть к голове.
    for rev != nil {
        // сравниваем значения двух списков, если не равны, то возвращаем
        // отрицание
        if head.Val != rev.Val {
            return false
        }
        // двигаемся по спискам дальше
        head = head.Next
        rev = rev.Next
    }
    return true
}
```

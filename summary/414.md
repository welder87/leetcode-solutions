# 414. Third Maximum Number

## [Description](https://leetcode.com/problems/third-maximum-number/)

## Solution

```go
// Ключевая идея:
// Используем Ordered Set для хранения уникальных элементов в порядке убывания.
// Time Complexity: O(n)
// Space Complexity: O(1) или o(3)
func thirdMax(nums []int) int {
    os := GetOrderedSet()
    for _, num := range nums {
        os.Push(num)
    }
    if third := os.Third(); third == math.MinInt64 {
        return os.First()
    } else {
        return third
    }
}

type OrderedSet struct{
    nums [3]int
}

func GetOrderedSet() *OrderedSet {
    // используем Sentinel Value для инициализации массива
    return &OrderedSet{nums: [3]int{math.MinInt64, math.MinInt64, math.MinInt64}}
}

func (s *OrderedSet) Push(val int) {
    if s.nums[0] == val || s.nums[1] == val || s.nums[2] == val{
        return
    }
    if s.nums[0] < val {
        s.nums[2], s.nums[1], s.nums[0] = s.nums[1], s.nums[0], val
    } else if s.nums[1] < val {
        s.nums[2], s.nums[1] = s.nums[1], val
    } else if s.nums[2] < val {
        s.nums[2] = val
    }
}

func (s *OrderedSet) Third() int {
    return s.nums[2]
}

func (s *OrderedSet) First() int {
    return s.nums[0]
}
```
